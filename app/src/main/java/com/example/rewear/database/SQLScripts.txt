CREATE TABLE User (
    id serial AUTO_INCREMENT PRIMARY KEY,
    first_name VARCHAR(15) NOT NULL,
    last_name VARCHAR(15) NOT NULL,
    username VARCHAR(15) NOT NULL,
    password VARCHAR(20) NOT NULL,
    UNIQUE (Username,id)
);

CREATE TABLE BelongsTo (
    group_id INT NOT NULL AUTO_INCREMENT,
    username VARCHAR(15) NOT NULL,

    FOREIGN KEY (username)
        REFERENCES User
        ON UPDATE CASCADE ON DELETE CASCADE
);

CREATE TABLE Group (
    group_id INT NOT NULL AUTO_INCREMENT,
    group_name VARCHAR(15) NOT NULL,
    group_desc VARCHAR(128) NOT NULL
);

CREATE TABLE PictureGroup (
    picture_id INT NOT NULL,
    group_id INT NOT NULL,
    username VARCHAR(15) NOT NULL

    FOREIGN KEY (username)
        REFERENCES User
        ON UPDATE CASCADE ON DELETE CASCADE
);

CREATE TABLE Picture (
    picture_id INT NOT NULL AUTO_INCREMENT,
    picture_path BLOB NOT NULL,
    caption VARCHAR(128) NOT NULL,
    date_posted DATE,

    FOREIGN KEY (picture_id)
        REFERENCES PictureGroup
        ON UPDATE CASCADE ON DELETE CASCADE
);

CREATE TABLE Comment (
    username VARCHAR(15) NOT NULL,
    picture_id INT NOT NULL
    comment_id INT NOT NULL AUTO_INCREMENT,
    comment VARCHAR(128) NOT NULL

    FOREIGN KEY (username)
        REFERENCES User
        ON UPDATE CASCADE ON DELETE CASCADE
);

CREATE TABLE Liked (
    username VARCHAR(15) NOT NULL,
    picture_id INT NOT NULL

    FOREIGN KEY (username)
        REFERENCES User
        ON UPDATE CASCADE ON DELETE CASCADE
);

CREATE TABLE Clothes (
    clothes_id INT NOT NULL AUTO_INCREMENT,
    clothes_pic BLOB NOT NULL,
    clothes_type VARCHAR(32) NOT NULL,
    clothes_desc VARCHAR(256) NOT NULL,
    username VARCHAR(15) NOT NULL,
    date_created DATE NOT NULL,
    date_added DATE NOT NULL

    FOREIGN KEY (username)
        REFERENCES User
        ON UPDATE CASCADE ON DELETE CASCADE
);

CREATE TABLE DateWorn (
    clothes_id INT NOT NULL,
    date_worn DATE

    FOREIGN KEY (clothes_id)
        REFERENCES Clothes
        ON UPDATE CASCADE ON DELETE CASCADE
);
